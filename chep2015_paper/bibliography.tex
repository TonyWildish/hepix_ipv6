\section*{References}

\begin{thebibliography}{1}
\bibitem{ipv6wg} {\tt http://hepix-ipv6.web.cern.ch}
\bibitem{ripeness} The status of IPv6 readiness of the local Internet Registries
that refer to RIPE is monitored in the ``IPv6 RIPEness'' pages ({\tt https://ipv6ripeness.ripe.net/} at the time of writing). 
\bibitem{ripestat} RIPE provides access to a comprehensive set of statistical
data via {\tt https://stat.ripe.net/data}. This data set is also used for
the online diagnostics and statistical tools (e.g. BGPlay). 
\bibitem{ripeipv6} Specific
IPv6 trend graphs for the RIPE LIRs can be accessed at {\tt https://labs.ripe.net/statistics/?tags=ipv6}.
\bibitem{arinstat} A collection of trend plots collected by ARIN
can be found here: {\tt https://www.arin.net/knowledge/statistics/}.
\bibitem{apnicstat} A somewhat out-of-date set of IPv6
statistics data gathered by APNIC can be found here: {\tt https://labs.apnic.net/ipv6-measurement/}.
\bibitem{lacnicipc6} The LACNIC IPv6 portal is at: {\tt http://portalipv6.lacnic.net/en/}.
\bibitem{afrinicipv6} A collection of AFRINIC IPv6 resources can be found here: {\tt http://www.afrinic.net/en/services/statistics/ipv6-resources}.
\bibitem{akamaistats} The historical statistics of IPv6 traffic on Akamai usare are recorded
at {\tt http://www.akamai.com/ipv6/} at the time of writing.
\bibitem{googlestats} The historical statistics of IPv6 client traffic on Google are recorded
at {\tt http://www.google.com/intl/en/ipv6/statistics.html} at the time of writing.
\bibitem{readiness} Our working group tracks the IPv6 readiness of applications
of interest to WLCT at {\tt http://hepix-ipv6.web.cern.ch/wlcg-applications}.
\bibitem{rfc} All Internet Engineering Task Force Requests For Comments (RFC) documents are available
from URLs such as http://www.ietf.org/rfc/rfcNNNN.txt where NNNN is the RFC number, for example {\tt http://www.ietf.org/rfc/rfc2460.txt}
\bibitem{ipv6stat} See for instance {\tt http://www.google.com/ipv6/statistics.html}. The 5\% global connectivity threshold was crossed in January 2015.
\bibitem{wdayperf} Plonka D, Barford P, Assessing performance of Internet services on IPv6, 2013 IEEE Symposium on Computers and Communications (ISCC), {\tt doi:10.1109/ISCC.2013.6755050}
\bibitem{FTS3}
    Salichos M, Keeble O, Alvarez Ayllon A, Kamil Simon M 2013 FTS3 - Robust, simplified and high-performance data movement service for WLCG {\it also presented at CHEP 2013}
\end{thebibliography}
